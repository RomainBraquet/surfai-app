async function loadAndRenderSpots() {
    const list = document.getElementById('spots-list');
    const btn = document.getElementById('spots-load-more');
    if (!list) return;

    if (_spotsPage === 0) list.innerHTML = '<div class="empty">Chargement...</div>';

    let query = supabaseClient
        .from('spots')
        .select('id, name, city, region, country, lat, lng, wave_type, difficulty')
        .range(_spotsPage * SPOTS_PAGE_SIZE, (_spotsPage + 1) * SPOTS_PAGE_SIZE - 1)
        .order('name');

    if (_spotsFilters.country) query = query.eq('country', _spotsFilters.country);
    if (_spotsFilters.region) query = query.eq('region', _spotsFilters.region);
    if (_spotsFilters.search && _spotsFilters.search.length >= 2) {
        const s = _spotsFilters.search.replace(/[.,()]/g, '');
        query = query.or(`name.ilike.%${s}%,city.ilike.%${s}%,region.ilike.%${s}%,country.ilike.%${s}%`);
    }

    const { data, error } = await query;
    const spots = data || [];

    if (_spotsPage === 0) {
        _spotsLoaded = spots;
        list.innerHTML = '';
    } else {
        _spotsLoaded.push(...spots);
    }

    // Sort client-side si distance
    let sorted = [..._spotsLoaded];
    if (_spotsSort === 'distance' && _userLat != null) {
        sorted.sort((a, b) => {
            const da = haversineKm(a.lat, a.lng, _userLat, _userLng);
            const db = haversineKm(b.lat, b.lng, _userLat, _userLng);
            return da - db;
        });
    }

    // Render
    if (sorted.length === 0) {
        list.innerHTML = '<div class="empty">Aucun spot trouvé.</div>';
    } else {
        list.innerHTML = sorted.map(s => {
            const waveType = s.wave_type ? s.wave_type.replace(/_/g, ' ') : '';
            const loc = [s.city, s.country].filter(Boolean).join(' · ');
            const meta = [loc, waveType].filter(Boolean).join(' · ');
            return `<div class="spot-row" onclick="showSpotDetail('${s.id}')">
                <div class="spot-row-info">
                    <div class="spot-row-name">${s.name}</div>
                    <div class="spot-row-meta">${meta}</div>
                </div>
            </div>`;
        }).join('');
    }

    // Voir plus
    if (btn) {
        btn.style.display = spots.length < SPOTS_PAGE_SIZE ? 'none' : 'block';
        btn.disabled = false;
        btn.textContent = 'Voir plus';
    }
}

function loadMoreSpots() {
    const btn = document.getElementById('spots-load-more');
    if (btn) { btn.disabled = true; btn.textContent = 'Chargement...'; }
    _spotsPage++;
    loadAndRenderSpots();
}

async function loadSpotsFilters() {
    const countrySelect = document.getElementById('spots-country');
    if (!countrySelect) return;
    const { data } = await supabaseClient
        .from('spots')
        .select('country')
        .not('country', 'is', null)
        .order('country');
    const countries = [...new Set((data || []).map(s => s.country))];
    countrySelect.innerHTML = '<option value="">Tous les pays</option>' +
        countries.map(c => `<option value="${c}">${c}</option>`).join('');
}

async function onSpotsCountryChange() {
    const country = document.getElementById('spots-country')?.value || '';
    const regionSelect = document.getElementById('spots-region');
    _spotsFilters.country = country || undefined;
    _spotsFilters.region = undefined;

    if (country && regionSelect) {
        const { data } = await supabaseClient
            .from('spots')
            .select('region')
            .eq('country', country)
            .not('region', 'is', null)
            .order('region');
        const regions = [...new Set((data || []).map(s => s.region))];
        regionSelect.innerHTML = '<option value="">Toutes les régions</option>' +
            regions.map(r => `<option value="${r}">${r}</option>`).join('');
    }

    renderActiveFilters();
    _spotsPage = 0; _spotsLoaded = [];
    loadAndRenderSpots();
}

function onSpotsRegionChange() {
    _spotsFilters.region = document.getElementById('spots-region')?.value || undefined;
    renderActiveFilters();
    _spotsPage = 0; _spotsLoaded = [];
    loadAndRenderSpots();
}

function renderActiveFilters() {
    const el = document.getElementById('active-filters');
    if (!el) return;
    const chips = [];
    if (_spotsFilters.country) chips.push(`<span class="active-filter" onclick="clearFilter('country')">✕ ${_spotsFilters.country}</span>`);
    if (_spotsFilters.region) chips.push(`<span class="active-filter" onclick="clearFilter('region')">✕ ${_spotsFilters.region}</span>`);
    el.innerHTML = chips.join('');
}

function clearFilter(key) {
    delete _spotsFilters[key];
    if (key === 'country') { delete _spotsFilters.region; document.getElementById('spots-country').value = ''; }
    if (key === 'region') document.getElementById('spots-region').value = '';
    renderActiveFilters();
    _spotsPage = 0; _spotsLoaded = [];
    loadAndRenderSpots();
}

function setSpotsSort(sort) {
    _spotsSort = sort;
    document.querySelectorAll('.sort-chip').forEach(c => c.classList.toggle('active', c.dataset.sort === sort));
    if (sort === 'filter') { toggleFilterPanel(); return; }
    _spotsPage = 0; _spotsLoaded = [];
    loadAndRenderSpots();
}

function toggleFilterPanel() {
    document.getElementById('filter-panel')?.classList.toggle('open');
}

function haversineKm(lat1, lng1, lat2, lng2) {
    const R = 6371;
    const toRad = d => d * Math.PI / 180;
    const dLat = toRad(lat2 - lat1);
    const dLng = toRad(lng2 - lng1);
    const a = Math.sin(dLat/2)**2 + Math.cos(toRad(lat1)) * Math.cos(toRad(lat2)) * Math.sin(dLng/2)**2;
    return R * 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
}
